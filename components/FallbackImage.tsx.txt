"use client";

import Image from "next/image";
import { useState } from "react";

interface FallbackImageProps {
  src: string;
  fallback: string;
  alt: string;
  width?: number;
  height?: number;
  className?: string;
}

export function FallbackImage({ src, fallback, alt, width, height, className }: FallbackImageProps) {
  const [imgSrc, setImgSrc] = useState(src);

  return (
    <Image
      src={imgSrc}
      alt={alt}
      width={width || 100}
      height={height || 100}
      className={className}
      onError={() => setImgSrc(fallback)}
    />
  );
}
